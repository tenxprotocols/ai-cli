# AI CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a static-binary Go CLI (`ai`) that talks to Anthropic, OpenAI, Gemini, OpenRouter, and OpenAI-compatible endpoints through a unified interface, with git-style subcommand dispatch, Unix-friendly I/O, and resumable chat sessions.

**Architecture:** Single Go binary with Cobra for subcommands. A thin `Provider` interface is implemented by adapters over the official SDKs (`anthropic-sdk-go`, `openai-go`, `google.golang.org/genai`). Config is TOML with profiles; chat history is append-only JSONL. Install-time symlinks (`ai-chat`, `ai-ask`, …) dispatch via `argv[0]`; unknown subcommands fall through to `ai-<name>` on `$PATH` (Git's extensibility model) or to the `ask` subcommand as a final fallback.

**Tech Stack:**
- Go 1.22+, `CGO_ENABLED=0` for static builds
- `github.com/spf13/cobra` — CLI framework
- `github.com/pelletier/go-toml/v2` — TOML config
- `github.com/anthropics/anthropic-sdk-go` — Anthropic
- `github.com/openai/openai-go/v3` — OpenAI + OpenRouter + openai-compat
- `google.golang.org/genai` — Gemini
- `github.com/charmbracelet/bubbletea` + `bubbles/textarea` — REPL input
- `github.com/stretchr/testify` — test assertions
- [Task](https://taskfile.dev) — build tool
- [release-please](https://github.com/googleapis/release-please) — release automation

```go
// GlobalFlags holds flags that apply to every subcommand. Bound on the root
// command so they're inherited; individual subcommands read via Context.
type GlobalFlags struct {
	Profile    string
	Provider   string
	Model      string
	Format     string
	NoStream   bool
	System     string
	SystemFile string
	ConfigPath string
}

// NewRoot builds the top-level `ai` command.
func NewRoot() *cobra.Command {
	gf := &GlobalFlags{}
	root := &cobra.Command{
		Use:           "ai",
		Short:         "Talk to LLMs from the command line",
		SilenceUsage:  true,
		SilenceErrors: true,
	}

	pf := root.PersistentFlags()
	pf.StringVar(&gf.Profile, "profile", "", "profile name (env: AI_CLI_PROFILE)")
	pf.StringVar(&gf.Provider, "provider", "", "override provider for this call")
	pf.StringVar(&gf.Model, "model", "", "override model (verbatim to provider)")
	pf.StringVar(&gf.Format, "format", "text", "output format: text|json|jsonl")
	pf.BoolVar(&gf.NoStream, "no-stream", false, "disable streaming")
	pf.StringVar(&gf.System, "system", "", "system prompt (inline)")
	pf.StringVar(&gf.SystemFile, "system-file", "", "system prompt from file")
	pf.StringVar(&gf.ConfigPath, "config", "", "config file path (env: AI_CLI_CONFIG)")
```

```go
func TestResolveArgs(t *testing.T) {
```

- [ ] **Step 3: Write `internal/cli/dispatch.go`**

```go
package cli

import (
	"os"
	"os/exec"
	"path/filepath"
	"runtime"
	"strings"
	"syscall"
)

type ResolutionKind int

const (
	ResolveBuiltin ResolutionKind = iota
	ResolvePlugin
	ResolveAskFallback
)

type Resolution struct {
	Kind       ResolutionKind
	Args       []string // args to hand to cobra (for Builtin/AskFallback) or child (for Plugin)
	PluginPath string
}

// PluginLookup finds `ai-<name>` on $PATH. Returns (absPath, true) if found.
type PluginLookup func(name string) (string, bool)

// DefaultPluginLookup uses exec.LookPath.
func DefaultPluginLookup(name string) (string, bool) {
	p, err := exec.LookPath("ai-" + name)
	if err != nil {
		return "", false
	}
	return p, true
}

// ResolveArgs applies the dispatch rules described in the spec.
//
//	argv0    — os.Args[0]'s basename
//	argv     — full os.Args
//	known    — set of Cobra-registered subcommands
//	lookup   — plugin lookup (pass nil to skip plugin dispatch)
//
// Rules:
//  1. If argv0 matches "ai-<name>" and <name> is a known subcommand, rewrite
//     to {name, ...rest}.
//  2. Else split argv into (globalFlags, first-non-flag, rest). If the first
//     non-flag arg is a known subcommand, pass through. Otherwise, if a plugin
//     exists for it, exec that. Otherwise, treat everything after the global
//     flags as an `ask` invocation: {...globalFlags, "ask", ...rest}.
//  3. `ai` alone, or with only global flags/--help, pass through unchanged.
func ResolveArgs(argv0 string, argv []string, known map[string]bool, lookup PluginLookup) Resolution {
	base := filepath.Base(argv0)

	// Rule 1: argv0 is ai-<name>.
	if strings.HasPrefix(base, "ai-") && base != "ai" {
		name := strings.TrimPrefix(base, "ai-")
		if known[name] {
			return Resolution{Kind: ResolveBuiltin, Args: append([]string{name}, argv[1:]...)}
		}
		// Unknown ai-<name> binary invoked directly — let cobra show the error.
		return Resolution{Kind: ResolveBuiltin, Args: argv[1:]}
	}

	// Rule 2/3: argv[0] is "ai" (or equivalent). Partition rest into
	// (leadingFlags, positionals).
	rest := argv[1:]
	leading := []string{}
	i := 0
	for i < len(rest) {
		a := rest[i]
		if !strings.HasPrefix(a, "-") {
			break
		}
		// --help and -h short-circuit to cobra.
		if a == "--help" || a == "-h" || a == "--version" || a == "-v" {
			return Resolution{Kind: ResolveBuiltin, Args: rest}
		}
		leading = append(leading, a)
		// If flag takes a value and uses space form, consume the next arg.
		if !strings.Contains(a, "=") && flagTakesValue(a) && i+1 < len(rest) {
			i++
			leading = append(leading, rest[i])
		}
		i++
	}
	positionals := rest[i:]

	if len(positionals) == 0 {
		return Resolution{Kind: ResolveBuiltin, Args: rest}
	}
	first, tail := positionals[0], positionals[1:]

	if known[first] {
		return Resolution{Kind: ResolveBuiltin, Args: rest}
	}
	if lookup != nil {
		if path, ok := lookup(first); ok {
			childArgs := append([]string{"ai-" + first}, tail...)
			return Resolution{Kind: ResolvePlugin, Args: childArgs, PluginPath: path}
		}
	}
	// Fallback: everything becomes `ai ask ...`.
	out := make([]string, 0, len(leading)+1+len(positionals))
	out = append(out, leading...)
	out = append(out, "ask")
	out = append(out, positionals...)
	return Resolution{Kind: ResolveAskFallback, Args: out}
}

// flagTakesValue reports whether a global flag consumes the next argv entry.
// Boolean flags (--no-stream) do not. Any non-boolean global flag does.
func flagTakesValue(flag string) bool {
	name := strings.TrimLeft(flag, "-")
	if idx := strings.Index(name, "="); idx >= 0 {
		name = name[:idx]
	}
	switch name {
	case "no-stream":
		return false
	default:
		return true
	}
}

// Exec hands control to a plugin binary. On Unix this replaces the process;
// on Windows, it spawns and relays the exit code.
func Exec(path string, args []string) error {
	if runtime.GOOS == "windows" {
		cmd := exec.Command(path, args[1:]...)
		cmd.Stdin = os.Stdin
		cmd.Stdout = os.Stdout
		cmd.Stderr = os.Stderr
		err := cmd.Run()
		if ee, ok := err.(*exec.ExitError); ok {
			os.Exit(ee.ExitCode())
		}
		return err
	}
	return syscall.Exec(path, args, os.Environ())
}
```

```go
package providers

import (
	"errors"
	"fmt"
	"sync"
)

// ErrUnknownProvider is returned when a requested provider name has no
// constructor registered.
var ErrUnknownProvider = errors.New("unknown provider")

// Config is passed to each provider constructor. Adapters ignore fields they
// don't understand.
type Config struct {
	Name    string            // user-facing name (config block key)
	Type    string            // built-in type: anthropic|openai|openrouter|gemini|openai-compat
	APIKey  string
	BaseURL string            // for openai-compat
	Extra   map[string]string // future-proofing for provider-specific config
}

// Constructor builds a Provider from a Config.
type Constructor func(Config) (Provider, error)

type Registry struct {
	mu    sync.RWMutex
	ctors map[string]Constructor // keyed by Type
}

func NewRegistry() *Registry {
	return &Registry{ctors: map[string]Constructor{}}
}

func (r *Registry) Register(typ string, c Constructor) {
	r.mu.Lock()
	defer r.mu.Unlock()
	r.ctors[typ] = c
}

// Get builds a Provider for the given type. cfg.Name is used as the
// provider's Name() (so the openai adapter can identify itself as
// "openrouter" or "ollama" based on how it was configured).
func (r *Registry) Get(typ string, cfg Config) (Provider, error) {
	r.mu.RLock()
	c, ok := r.ctors[typ]
	r.mu.RUnlock()
	if !ok {
		return nil, fmt.Errorf("%w: %s", ErrUnknownProvider, typ)
	}
	cfg.Type = typ
	return c(cfg)
}

// Types returns the registered type names, sorted.
func (r *Registry) Types() []string {
	r.mu.RLock()
	defer r.mu.RUnlock()
	out := make([]string, 0, len(r.ctors))
	for k := range r.ctors {
		out = append(out, k)
	}
	return out
}
```

```go
package config

import (
	"errors"
	"fmt"
	"io/fs"
	"os"

	toml "github.com/pelletier/go-toml/v2"
)

// LoadFile reads a TOML config. A missing file is not an error — an empty
// but initialized File is returned.
func LoadFile(path string) (File, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		if errors.Is(err, fs.ErrNotExist) {
			return File{
				Providers: map[string]Provider{},
				Profiles:  map[string]Profile{},
			}, nil
		}
		return File{}, fmt.Errorf("read %s: %w", path, err)
	}

	var f File
	if err := toml.Unmarshal(data, &f); err != nil {
		return File{}, fmt.Errorf("parse %s: %w", path, err)
	}
	if f.Providers == nil {
		f.Providers = map[string]Provider{}
	}
	if f.Profiles == nil {
		f.Profiles = map[string]Profile{}
	}
	return f, nil
}

// SaveFile writes the TOML back. Used by `ai config set`.
func SaveFile(path string, f File) error {
	data, err := toml.Marshal(f)
	if err != nil {
		return fmt.Errorf("marshal: %w", err)
	}
	return os.WriteFile(path, data, 0o600)
}

// DefaultPath returns the platform-appropriate default config file path.
func DefaultPath() (string, error) {
	dir, err := os.UserConfigDir()
	if err != nil {
		return "", err
	}
	return fmt.Sprintf("%s/ai-cli/config.toml", dir), nil
}
```

```go
package config

import (
	"errors"
	"fmt"
	"strings"
)

var (
	ErrUnknownProfile  = errors.New("unknown profile")
	ErrUnknownProvider = errors.New("unknown provider")
	ErrMissingAPIKey   = errors.New("missing API key")
)

// EnvLookup is an injectable env var reader, so tests don't depend on the real
// environment.
type EnvLookup func(string) (string, bool)

// OSEnv is the production EnvLookup.
func OSEnv(k string) (string, bool) {
	if v, ok := osLookupEnv(k); ok {
		return v, true
	}
	return "", false
}

// Overrides holds CLI flag values. Empty strings mean "not set".
type Overrides struct {
	Profile  string
	Provider string
	Model    string
	System   string
}

// Resolved is the fully resolved view used by subcommands.
type Resolved struct {
	Profile      string
	ProviderName string // config block key
	ProviderType string // built-in type
	BaseURL      string
	APIKey       string
	Model        string
	System       string
	Temperature  *float64
	MaxTokens    *int
}

// Resolve applies precedence: flag > AI_CLI_* env > public env > file.
func Resolve(f File, o Overrides, env EnvLookup) (Resolved, error) {
	// 1. Profile name.
	profileName := firstNonEmpty(o.Profile, envOr(env, "AI_CLI_PROFILE"), f.DefaultProfile)
	if profileName == "" {
		return Resolved{}, fmt.Errorf("%w: no profile selected", ErrUnknownProfile)
	}
	prof, ok := f.Profiles[profileName]
	if !ok {
		return Resolved{}, fmt.Errorf("%w: %s", ErrUnknownProfile, profileName)
	}

	// 2. Provider name.
	provName := firstNonEmpty(o.Provider, envOr(env, "AI_CLI_PROVIDER"), prof.Provider)
	provCfg, ok := f.Providers[provName]
	if !ok {
		return Resolved{}, fmt.Errorf("%w: %s", ErrUnknownProvider, provName)
	}

	// 3. Model — verbatim.
	model := firstNonEmpty(o.Model, envOr(env, "AI_CLI_MODEL"), prof.Model)
	if model == "" {
		return Resolved{}, fmt.Errorf("no model selected for profile %q", profileName)
	}

	// 4. System prompt.
	system := firstNonEmpty(o.System, envOr(env, "AI_CLI_SYSTEM"), prof.System)

	// 5. API key: AI_CLI_<NAME>_API_KEY > public(type) > file.
	apiKey := resolveAPIKey(provName, provCfg.Type, provCfg.APIKey, env)

	return Resolved{
		Profile:      profileName,
		ProviderName: provName,
		ProviderType: provCfg.Type,
		BaseURL:      provCfg.BaseURL,
		APIKey:       apiKey,
		Model:        model,
		System:       system,
		Temperature:  prof.Temperature,
		MaxTokens:    prof.MaxTokens,
	}, nil
}

func resolveAPIKey(name, typ, fileVal string, env EnvLookup) string {
	if v, ok := env("AI_CLI_" + strings.ToUpper(name) + "_API_KEY"); ok && v != "" {
		return v
	}
	switch typ {
	case "anthropic":
		if v, ok := env("ANTHROPIC_API_KEY"); ok && v != "" {
			return v
		}
	case "openai":
		if v, ok := env("OPENAI_API_KEY"); ok && v != "" {
			return v
		}
	case "openrouter":
		if v, ok := env("OPENROUTER_API_KEY"); ok && v != "" {
			return v
		}
	case "gemini":
		if v, ok := env("GEMINI_API_KEY"); ok && v != "" {
			return v
		}
		if v, ok := env("GOOGLE_API_KEY"); ok && v != "" {
			return v
		}
	}
	return fileVal
}

func firstNonEmpty(xs ...string) string {
	for _, x := range xs {
		if x != "" {
			return x
		}
	}
	return ""
}

func envOr(env EnvLookup, key string) string {
	if v, ok := env(key); ok {
		return v
	}
	return ""
}
```

```go
// GetKey returns a value from the TOML tree by dotted path. Only known paths
// are supported.
func GetKey(f File, key string) (string, error) {
	switch key {
	case "default_profile":
		return f.DefaultProfile, nil
	}
	// providers.<name>.{type|api_key|base_url}
	if n, field, ok := splitThree(key, "providers."); ok {
		p, ok := f.Providers[n]
		if !ok {
			return "", fmt.Errorf("unknown provider: %s", n)
		}
		switch field {
		case "type":
			return p.Type, nil
		case "api_key":
			return p.APIKey, nil
		case "base_url":
			return p.BaseURL, nil
		}
	}
	// profiles.<name>.{provider|model|system}
	if n, field, ok := splitThree(key, "profiles."); ok {
		p, ok := f.Profiles[n]
		if !ok {
			return "", fmt.Errorf("unknown profile: %s", n)
		}
		switch field {
		case "provider":
			return p.Provider, nil
		case "model":
			return p.Model, nil
		case "system":
			return p.System, nil
		}
	}
	return "", fmt.Errorf("unsupported key: %s", key)
}

// SetKey writes a value by dotted path. Creates parents as needed.
func SetKey(f *File, key, val string) error {
	switch key {
	case "default_profile":
		f.DefaultProfile = val
		return nil
	}
	if n, field, ok := splitThree(key, "providers."); ok {
		p := f.Providers[n]
		switch field {
		case "type":
			p.Type = val
		case "api_key":
			p.APIKey = val
		case "base_url":
			p.BaseURL = val
		default:
			return fmt.Errorf("unsupported key: %s", key)
		}
		if f.Providers == nil {
			f.Providers = map[string]Provider{}
		}
		f.Providers[n] = p
		return nil
	}
	if n, field, ok := splitThree(key, "profiles."); ok {
		p := f.Profiles[n]
		switch field {
		case "provider":
			p.Provider = val
		case "model":
			p.Model = val
		case "system":
			p.System = val
		default:
			return fmt.Errorf("unsupported key: %s", key)
		}
		if f.Profiles == nil {
			f.Profiles = map[string]Profile{}
		}
		f.Profiles[n] = p
		return nil
	}
	return fmt.Errorf("unsupported key: %s", key)
}

// splitThree splits "prefix.<name>.<field>" into (name, field, ok).
func splitThree(s, prefix string) (string, string, bool) {
	if !strings.HasPrefix(s, prefix) {
		return "", "", false
	}
	rest := s[len(prefix):]
	dot := strings.IndexByte(rest, '.')
	if dot < 0 {
		return "", "", false
	}
	return rest[:dot], rest[dot+1:], true
}
```

```go
type Format string

const (
	FormatText  Format = "text"
	FormatJSON  Format = "json"
	FormatJSONL Format = "jsonl"
)

// ParseFormat returns an error on unknown values.
func ParseFormat(s string) (Format, error) {
	switch Format(s) {
	case FormatText, FormatJSON, FormatJSONL:
		return Format(s), nil
	}
	return "", fmt.Errorf("unknown format: %s (expected text|json|jsonl)", s)
}

// Writer consumes a stream of Chunks and writes the chosen format.
type Writer interface {
	Consume(ch <-chan providers.Chunk) error
	Finish() error
}

// NewWriter selects the concrete writer.
func NewWriter(f Format, w io.Writer) Writer {
	switch f {
	case FormatText:
		return &textWriter{w: w}
	case FormatJSON:
		return newJSONWriter(w)
	case FormatJSONL:
		return newJSONLWriter(w)
	}
	return &textWriter{w: w}
```

```go
import (
	"fmt"
	"io"

	"github.com/tenxprotocols/ai-cli/internal/providers"

type textWriter struct {
	w         io.Writer
	wroteText bool
	toolCalls int
	err       error
}

func (t *textWriter) Consume(ch <-chan providers.Chunk) error {
	for c := range ch {
		switch c.Type {
		case providers.ChunkTextDelta:
			if _, err := io.WriteString(t.w, c.Text); err != nil {
				return err
			}
			if c.Text != "" {
				t.wroteText = true
			}
		case providers.ChunkToolCallStart:
			t.toolCalls++
		case providers.ChunkError:
			t.err = c.Err
			return c.Err
		}
	}
	return nil
}

func (t *textWriter) Finish() error {
	if t.err != nil {
		return t.err
	}
	if t.wroteText {
		fmt.Fprintln(t.w)
	}
	if t.toolCalls > 0 {
		fmt.Fprintf(t.w, "<tool-calls: %d>\n", t.toolCalls)
	}
	return nil
}
```

```go
import (
	"encoding/json"
	"io"

	"github.com/tenxprotocols/ai-cli/internal/providers"
)

type jsonWriter struct {
	w          io.Writer
	text       []byte
	toolCalls  []providers.ToolCall
	usage      *providers.Usage
	stopReason string
	err        error
}

func newJSONWriter(w io.Writer) *jsonWriter { return &jsonWriter{w: w} }

func (j *jsonWriter) Consume(ch <-chan providers.Chunk) error {
	for c := range ch {
		switch c.Type {
		case providers.ChunkTextDelta:
			j.text = append(j.text, c.Text...)
		case providers.ChunkToolCallStart:
			if c.ToolCall != nil {
				j.toolCalls = append(j.toolCalls, *c.ToolCall)
			}
		case providers.ChunkUsage:
			j.usage = c.Usage
		case providers.ChunkMessageStop:
			if j.stopReason == "" {
				j.stopReason = providers.StopEndTurn
			}
		case providers.ChunkError:
			j.err = c.Err
			return c.Err
		}
	}
	if j.stopReason == "" {
		j.stopReason = providers.StopEndTurn
	}
	if len(j.toolCalls) > 0 && j.stopReason == providers.StopEndTurn {
		j.stopReason = providers.StopToolUse
	}
	return nil
}

func (j *jsonWriter) Finish() error {
	payload := map[string]any{
		"text":        string(j.text),
		"tool_calls":  j.toolCalls,
		"stop_reason": j.stopReason,
		"usage":       j.usage,
	}
	if j.err != nil {
		payload["error"] = j.err.Error()
	}
	enc := json.NewEncoder(j.w)
	enc.SetEscapeHTML(false)
	return enc.Encode(payload)
}
```

```go
	enc.SetEscapeHTML(false)

func (j *jsonlWriter) Consume(ch <-chan providers.Chunk) error {
	for c := range ch {
		obj := map[string]any{"type": c.Type}
		if c.Text != "" {
			obj["text"] = c.Text
		}
		if c.ToolCall != nil {
			obj["tool_call"] = c.ToolCall
		}
		if c.Usage != nil {
			obj["usage"] = c.Usage
		}
		if c.Err != nil {
			obj["error"] = c.Err.Error()
			j.err = c.Err
		}
		if err := j.enc.Encode(obj); err != nil {
			return err
		}
		if j.err != nil {
			return j.err
		}
	}
	return nil
}

func (j *jsonlWriter) Finish() error {
	return j.enc.Encode(map[string]any{"type": "stop"})
```

```go
	defer s.mu.Unlock()
```

```go
package providers

import (
	"context"
	"encoding/json"
	"fmt"

	"github.com/openai/openai-go/v3"
	"github.com/openai/openai-go/v3/option"
)

// NewOpenAI constructs an adapter over the openai-go SDK. It backs three
// Config types: "openai", "openrouter", and "openai-compat".
func NewOpenAI(cfg Config) (Provider, error) {
	opts := []option.RequestOption{option.WithAPIKey(cfg.APIKey)}
	if cfg.BaseURL != "" {
		opts = append(opts, option.WithBaseURL(cfg.BaseURL))
	}
	if cfg.Type == "openrouter" {
		opts = append(opts,
			option.WithHeader("HTTP-Referer", "https://github.com/tenxprotocols/ai-cli"),
			option.WithHeader("X-Title", "ai-cli"),
		)
	}
	client := openai.NewClient(opts...)
	return &openaiProvider{name: cfg.Name, client: client}, nil
}

type openaiProvider struct {
	name   string
	client openai.Client
}

func (p *openaiProvider) Name() string { return p.name }

func (p *openaiProvider) Complete(ctx context.Context, req Request) (Response, error) {
	params, err := p.buildParams(req)
	if err != nil {
		return Response{}, err
	}
	completion, err := p.client.Chat.Completions.New(ctx, params)
	if err != nil {
		return Response{}, err
	}
	if len(completion.Choices) == 0 {
		return Response{}, fmt.Errorf("openai: empty response")
	}
	choice := completion.Choices[0]

	msg := Message{Role: RoleAssistant}
	if choice.Message.Content != "" {
		msg.Content = append(msg.Content, ContentPart{Type: PartText, Text: choice.Message.Content})
	}
	for _, tc := range choice.Message.ToolCalls {
		msg.ToolCalls = append(msg.ToolCalls, ToolCall{
			ID: tc.ID, Name: tc.Function.Name,
			Arguments: json.RawMessage(tc.Function.Arguments),
		})
	}

	stop := StopEndTurn
	switch choice.FinishReason {
	case "stop":
		stop = StopEndTurn
	case "length":
		stop = StopMaxTokens
	case "tool_calls", "function_call":
		stop = StopToolUse
	case "content_filter":
		stop = StopError
	}

	return Response{
		Messages: []Message{msg},
		Usage: Usage{
			InputTokens:  int(completion.Usage.PromptTokens),
			OutputTokens: int(completion.Usage.CompletionTokens),
		},
		StopReason: stop,
	}, nil
}

func (p *openaiProvider) Stream(ctx context.Context, req Request) (<-chan Chunk, error) {
	params, err := p.buildParams(req)
	if err != nil {
		return nil, err
	}
	stream := p.client.Chat.Completions.NewStreaming(ctx, params)

	ch := make(chan Chunk, 16)
	go func() {
		defer close(ch)
		for stream.Next() {
			evt := stream.Current()
			if len(evt.Choices) == 0 {
				continue
			}
			delta := evt.Choices[0].Delta
			if delta.Content != "" {
				ch <- Chunk{Type: ChunkTextDelta, Text: delta.Content}
			}
			for _, tc := range delta.ToolCalls {
				ch <- Chunk{Type: ChunkToolCallStart, ToolCall: &ToolCall{
					ID: tc.ID, Name: tc.Function.Name,
					Arguments: json.RawMessage(tc.Function.Arguments),
				}}
			}
		}
		if err := stream.Err(); err != nil {
			ch <- Chunk{Type: ChunkError, Err: err}
			return
		}
		ch <- Chunk{Type: ChunkMessageStop}
	}()
	return ch, nil
}

func (p *openaiProvider) ListModels(ctx context.Context) ([]ModelInfo, error) {
	page, err := p.client.Models.List(ctx)
	if err != nil {
		return nil, err
	}
	var out []ModelInfo
	for _, m := range page.Data {
		out = append(out, ModelInfo{ID: m.ID, DisplayName: m.ID})
	}
	return out, nil
}

func (p *openaiProvider) buildParams(req Request) (openai.ChatCompletionNewParams, error) {
	var msgs []openai.ChatCompletionMessageParamUnion
	if req.System != "" {
		msgs = append(msgs, openai.SystemMessage(req.System))
	}
	for _, m := range req.Messages {
		switch m.Role {
		case RoleUser:
			msgs = append(msgs, openai.UserMessage(flattenText(m.Content)))
		case RoleAssistant:
			msgs = append(msgs, openai.AssistantMessage(flattenText(m.Content)))
		case RoleTool:
			msgs = append(msgs, openai.ToolMessage(flattenText(m.Content), m.ToolCallID))
		case RoleSystem:
			msgs = append(msgs, openai.SystemMessage(flattenText(m.Content)))
		}
	}

	params := openai.ChatCompletionNewParams{
		Model:    req.Model,
		Messages: msgs,
	}
	if req.MaxTokens != nil {
		params.MaxCompletionTokens = openai.Int(int64(*req.MaxTokens))
	}
	if req.Temperature != nil {
		params.Temperature = openai.Float(*req.Temperature)
	}
	for _, t := range req.Tools {
		params.Tools = append(params.Tools,
			openai.ChatCompletionFunctionTool(openai.FunctionDefinitionParam{
				Name:        t.Name,
				Description: openai.String(t.Description),
				Parameters:  openai.FunctionParameters(mustJSONObject(t.Schema)),
			}),
		)
	}
	return params, nil
}

func flattenText(parts []ContentPart) string {
	var s string
	for _, p := range parts {
		if p.Type == PartText {
			s += p.Text
		}
	}
	return s
}

func mustJSONObject(raw json.RawMessage) map[string]any {
	if len(raw) == 0 {
		return map[string]any{"type": "object"}
	}
	var m map[string]any
	if err := json.Unmarshal(raw, &m); err != nil {
		return map[string]any{"type": "object"}
	}
	return m
}
```

```go
func NewAnthropic(cfg Config) (Provider, error) {
	opts := []option.RequestOption{option.WithAPIKey(cfg.APIKey)}
	if cfg.BaseURL != "" {
		opts = append(opts, option.WithBaseURL(cfg.BaseURL))
	}
	client := anthropic.NewClient(opts...)
	return &anthropicProvider{name: cfg.Name, client: client}, nil
}

type anthropicProvider struct {
	name   string
	client anthropic.Client
}

func (p *anthropicProvider) Name() string { return p.name }

func (p *anthropicProvider) Complete(ctx context.Context, req Request) (Response, error) {
	params, err := p.buildParams(req)
	if err != nil {
		return Response{}, err
	}
	msg, err := p.client.Messages.New(ctx, params)
	if err != nil {
		return Response{}, err
	}

	out := Message{Role: RoleAssistant}
	for _, block := range msg.Content {
		switch v := block.AsAny().(type) {
		case anthropic.TextBlock:
			out.Content = append(out.Content, ContentPart{Type: PartText, Text: v.Text})
		case anthropic.ToolUseBlock:
			out.ToolCalls = append(out.ToolCalls, ToolCall{
				ID: block.ID, Name: block.Name,
				Arguments: json.RawMessage(v.JSON.Input.Raw()),
			})
		}
	}

	stop := mapAnthropicStop(string(msg.StopReason))
	return Response{
		Messages: []Message{out},
		Usage: Usage{
			InputTokens:  int(msg.Usage.InputTokens),
			OutputTokens: int(msg.Usage.OutputTokens),
		},
		StopReason: stop,
	}, nil
}

func (p *anthropicProvider) Stream(ctx context.Context, req Request) (<-chan Chunk, error) {
	params, err := p.buildParams(req)
	if err != nil {
		return nil, err
	}
	stream := p.client.Messages.NewStreaming(ctx, params)

	ch := make(chan Chunk, 16)
	go func() {
		defer close(ch)
		for stream.Next() {
			evt := stream.Current()
			switch variant := evt.AsAny().(type) {
			case anthropic.ContentBlockDeltaEvent:
				switch d := variant.Delta.AsAny().(type) {
				case anthropic.TextDelta:
					ch <- Chunk{Type: ChunkTextDelta, Text: d.Text}
				}
			case anthropic.MessageStopEvent:
				ch <- Chunk{Type: ChunkMessageStop}
			}
		}
		if err := stream.Err(); err != nil {
			ch <- Chunk{Type: ChunkError, Err: err}
		}
	}()
	return ch, nil
}

func (p *anthropicProvider) buildParams(req Request) (anthropic.MessageNewParams, error) {
	var maxTokens int64 = 4096
	if req.MaxTokens != nil {
		maxTokens = int64(*req.MaxTokens)
	}

	var msgs []anthropic.MessageParam
	for _, m := range req.Messages {
		switch m.Role {
		case RoleUser:
			msgs = append(msgs, anthropic.NewUserMessage(anthropic.NewTextBlock(flattenText(m.Content))))
		case RoleAssistant:
			msgs = append(msgs, anthropic.NewAssistantMessage(anthropic.NewTextBlock(flattenText(m.Content))))
		case RoleTool:
			msgs = append(msgs, anthropic.NewUserMessage(anthropic.NewToolResultBlock(m.ToolCallID, flattenText(m.Content), false)))
		}
	}

	params := anthropic.MessageNewParams{
		Model:     anthropic.Model(req.Model),
		MaxTokens: maxTokens,
		Messages:  msgs,
	}
	if req.System != "" {
		params.System = []anthropic.TextBlockParam{{Text: req.System}}
	}
	if req.Temperature != nil {
		params.Temperature = anthropic.Float(*req.Temperature)
	}
	for _, t := range req.Tools {
		tp := anthropic.ToolParam{
			Name:        t.Name,
			Description: anthropic.String(t.Description),
		}
		var schema map[string]any
		_ = json.Unmarshal(t.Schema, &schema)
		props, _ := schema["properties"].(map[string]any)
		tp.InputSchema = anthropic.ToolInputSchemaParam{Properties: props}
		params.Tools = append(params.Tools, anthropic.ToolUnionParam{OfTool: &tp})
	}
	return params, nil
}

func mapAnthropicStop(s string) string {
	switch s {
	case "end_turn":
		return StopEndTurn
	case "max_tokens":
		return StopMaxTokens
	case "tool_use":
		return StopToolUse
	case "stop_sequence":
		return StopStopSequence
	default:
		return StopEndTurn
	}
}
```

```go
package providers

import (
	"context"
	"fmt"

	"google.golang.org/genai"
)

func NewGemini(cfg Config) (Provider, error) {
	clientCfg := &genai.ClientConfig{
		APIKey:  cfg.APIKey,
		Backend: genai.BackendGeminiAPI,
	}
	if cfg.BaseURL != "" {
		clientCfg.HTTPOptions = genai.HTTPOptions{BaseURL: cfg.BaseURL}
	}
	client, err := genai.NewClient(context.Background(), clientCfg)
	if err != nil {
		return nil, fmt.Errorf("gemini client: %w", err)
	}
	return &geminiProvider{name: cfg.Name, client: client}, nil
}

type geminiProvider struct {
	name   string
	client *genai.Client
}

func (p *geminiProvider) Name() string { return p.name }

func (p *geminiProvider) Complete(ctx context.Context, req Request) (Response, error) {
	contents := toGeminiContents(req.Messages)
	cfg := &genai.GenerateContentConfig{}
	if req.System != "" {
		cfg.SystemInstruction = &genai.Content{Parts: []*genai.Part{{Text: req.System}}}
	}
	if req.Temperature != nil {
		t := float32(*req.Temperature)
		cfg.Temperature = &t
	}
	if req.MaxTokens != nil {
		n := int32(*req.MaxTokens)
		cfg.MaxOutputTokens = n
	}

	resp, err := p.client.Models.GenerateContent(ctx, req.Model, contents, cfg)
	if err != nil {
		return Response{}, err
	}
	if len(resp.Candidates) == 0 {
		return Response{}, fmt.Errorf("gemini: empty response")
	}
	cand := resp.Candidates[0]
	out := Message{Role: RoleAssistant}
	if cand.Content != nil {
		for _, part := range cand.Content.Parts {
			if part.Text != "" {
				out.Content = append(out.Content, ContentPart{Type: PartText, Text: part.Text})
			}
		}
	}

	stop := StopEndTurn
	if cand.FinishReason == genai.FinishReasonMaxTokens {
		stop = StopMaxTokens
	}

	r := Response{Messages: []Message{out}, StopReason: stop}
	if resp.UsageMetadata != nil {
		r.Usage = Usage{
			InputTokens:  int(resp.UsageMetadata.PromptTokenCount),
			OutputTokens: int(resp.UsageMetadata.CandidatesTokenCount),
		}
	}
	return r, nil
}

func (p *geminiProvider) Stream(ctx context.Context, req Request) (<-chan Chunk, error) {
	contents := toGeminiContents(req.Messages)
	cfg := &genai.GenerateContentConfig{}
	if req.System != "" {
		cfg.SystemInstruction = &genai.Content{Parts: []*genai.Part{{Text: req.System}}}
	}

	iter := p.client.Models.GenerateContentStream(ctx, req.Model, contents, cfg)

	ch := make(chan Chunk, 16)
	go func() {
		defer close(ch)
		for resp, err := range iter {
			if err != nil {
				ch <- Chunk{Type: ChunkError, Err: err}
				return
			}
			if len(resp.Candidates) == 0 {
				continue
			}
			for _, part := range resp.Candidates[0].Content.Parts {
				if part.Text != "" {
					ch <- Chunk{Type: ChunkTextDelta, Text: part.Text}
				}
			}
		}
		ch <- Chunk{Type: ChunkMessageStop}
	}()
	return ch, nil
}

func (p *geminiProvider) ListModels(_ context.Context) ([]ModelInfo, error) {
	return []ModelInfo{
		{ID: "gemini-2.0-flash", DisplayName: "Gemini 2.0 Flash"},
		{ID: "gemini-2.5-pro", DisplayName: "Gemini 2.5 Pro"},
	}, nil
}

func toGeminiContents(msgs []Message) []*genai.Content {
	out := make([]*genai.Content, 0, len(msgs))
	for _, m := range msgs {
		role := "user"
		if m.Role == RoleAssistant {
			role = "model"
		}
		var parts []*genai.Part
		for _, c := range m.Content {
			if c.Type == PartText {
				parts = append(parts, &genai.Part{Text: c.Text})
			}
		}
		if len(parts) > 0 {
			out = append(out, &genai.Content{Role: role, Parts: parts})
		}
	}
	return out
}
```

```go
func WithInjectedProvider(ctx context.Context, name string, p providers.Provider) context.Context {
	m, _ := ctx.Value(injectedProvidersKey).(map[string]providers.Provider)

	m, ok := ctx.Value(injectedProvidersKey).(map[string]providers.Provider)
	if !ok {
		return nil, false
	}
	p, ok := m[name]
	return p, ok
```

```go
import (

func newAskCmd(gf *GlobalFlags) *cobra.Command {
	var (
		files         []string
		images        []string
		toolsPath     string
		conversation  string
	)
	c := &cobra.Command{
		Use:                   "ask [prompt words...]",
		Short:                 "Ask the model a single question",
		DisableFlagsInUseLine: true,
		SilenceUsage:          true,
		RunE: func(cmd *cobra.Command, args []string) error {
			ctx := cmd.Context()

			prompt := strings.Join(args, " ")
			if piped, ok := readStdinIfPiped(); ok {
				if prompt != "" {
					prompt = piped + "\n\n" + prompt
				} else {
					prompt = piped
				}
			}
			if prompt == "" && toolsPath == "" && conversation == "" {
				return fmt.Errorf("empty prompt: provide words, pipe stdin, or pass --conversation")
			}

			resolved, err := resolveForCall(gf)
			if err != nil {
				return err
			}

			prov, err := buildProvider(ctx, resolved)
			if err != nil {
				return err
			}

			req := providers.Request{
				Model:       resolved.Model,
				System:      resolved.System,
				Temperature: resolved.Temperature,
				MaxTokens:   resolved.MaxTokens,
				Messages: []providers.Message{{
					Role:    providers.RoleUser,
					Content: []providers.ContentPart{{Type: providers.PartText, Text: prompt}},
				}},
			}
			// Files/images/tools/conversation wired in later tasks.
			_ = files
			_ = images
			_ = toolsPath
			_ = conversation

			fmt, err := output.ParseFormat(gf.Format)
			if err != nil {
				return err
			}
			if gf.NoStream || fmt == output.FormatJSON {
				return runNonStream(ctx, prov, req, fmt, cmd.OutOrStdout())
			}
			return runStream(ctx, prov, req, fmt, cmd.OutOrStdout())
		},
	}
	c.Flags().StringSliceVar(&files, "file", nil, "attach file (repeatable)")
	c.Flags().StringSliceVar(&images, "image", nil, "attach image (repeatable)")
	c.Flags().StringVar(&toolsPath, "tools", "", "tool definitions (JSON)")
	c.Flags().StringVar(&conversation, "conversation", "", "continue from a JSONL conversation file")
	return c
}

func runStream(ctx context.Context, p providers.Provider, req providers.Request, f output.Format, out *os.File) error {
	return runStreamW(ctx, p, req, f, out)
}

func runStreamW(ctx context.Context, p providers.Provider, req providers.Request, f output.Format, w interface{ Write([]byte) (int, error) }) error {
	req.Stream = true
	ch, err := p.Stream(ctx, req)
	if err != nil {
		return err
	}
	wr := output.NewWriter(f, w)
	if err := wr.Consume(ch); err != nil {
		return err
	}
	return wr.Finish()
}

func runNonStream(ctx context.Context, p providers.Provider, req providers.Request, f output.Format, w interface{ Write([]byte) (int, error) }) error {
	req.Stream = false
	resp, err := p.Complete(ctx, req)
	if err != nil {
		return err
	}
	// Convert Response back into a synthetic Chunk stream so formatters are uniform.
	ch := make(chan providers.Chunk, 4)
	go func() {
		defer close(ch)
		for _, m := range resp.Messages {
			for _, part := range m.Content {
				if part.Type == providers.PartText && part.Text != "" {
					ch <- providers.Chunk{Type: providers.ChunkTextDelta, Text: part.Text}
				}
			}
			for _, tc := range m.ToolCalls {
				tc := tc
				ch <- providers.Chunk{Type: providers.ChunkToolCallStart, ToolCall: &tc}
			}
		}
		ch <- providers.Chunk{Type: providers.ChunkUsage, Usage: &resp.Usage}
		ch <- providers.Chunk{Type: providers.ChunkMessageStop}
	}()
	wr := output.NewWriter(f, w)
	if err := wr.Consume(ch); err != nil {
		return err
	}
	return wr.Finish()
}

func readStdinIfPiped() (string, bool) {
	fi, err := os.Stdin.Stat()
	if err != nil {
		return "", false
	}
	if (fi.Mode() & os.ModeCharDevice) != 0 {
		return "", false // interactive TTY
	}
	// Read entire stdin (reasonable for CLI prompts; no streaming-in yet).
	b, err := io.ReadAll(os.Stdin)
	if err != nil || len(b) == 0 {
		return "", false
	}
	return strings.TrimRight(string(b), "\n"), true
}

// resolveForCall resolves profile/provider/model for this invocation.
func resolveForCall(gf *GlobalFlags) (config.Resolved, error) {
	path, err := resolveConfigPath(gf.ConfigPath)
	if err != nil {
		return config.Resolved{}, err
	}
	f, err := config.LoadFile(path)
	if err != nil {
		return config.Resolved{}, err
	}
	overrides := config.Overrides{
		Profile: gf.Profile, Provider: gf.Provider,
		Model: gf.Model, System: readSystemPrompt(gf),
	}
	return config.Resolve(f, overrides, config.OSEnv)
}

func readSystemPrompt(gf *GlobalFlags) string {
	if gf.System != "" {
		return gf.System
	}
	if gf.SystemFile == "" {
		return ""
	}
	b, err := os.ReadFile(gf.SystemFile)
	if err != nil {
		return "" // caller surfaces any downstream error on send
	}
	return string(b)
}

// buildProvider returns the resolved provider, honoring test injection.
func buildProvider(ctx context.Context, r config.Resolved) (providers.Provider, error) {
	if p, ok := injectedProvider(ctx, r.ProviderName); ok {
		return p, nil
	}
	if r.APIKey == "" {
		return nil, fmt.Errorf("no API key for provider %q (set AI_CLI_%s_API_KEY or config)",
			r.ProviderName, strings.ToUpper(r.ProviderName))
	}
	reg := providers.NewRegistry()
	providers.RegisterBuiltins(reg)
	return reg.Get(r.ProviderType, providers.Config{
		Name: r.ProviderName, APIKey: r.APIKey, BaseURL: r.BaseURL,
	})
}
```

```go
func buildUserMessage(prompt string, files, images []string) (providers.Message, error) {
	parts := []providers.ContentPart{}

	for _, f := range files {
		b, err := os.ReadFile(f)
		if err != nil {
			return providers.Message{}, fmt.Errorf("read %s: %w", f, err)
		}
		if !looksLikeText(b) {
			return providers.Message{}, fmt.Errorf("%s is binary — use --image instead", f)
		}
		parts = append(parts, providers.ContentPart{
			Type: providers.PartText,
			Text: fmt.Sprintf("# File: %s\n\n%s", f, string(b)),
		})
	}

	for _, img := range images {
		b, err := os.ReadFile(img)
		if err != nil {
			return providers.Message{}, fmt.Errorf("read %s: %w", img, err)
		}
		mime := mimeFromExt(img)
		parts = append(parts, providers.ContentPart{Type: providers.PartImage, MIMEType: mime, Data: b})
	}

	if prompt != "" {
		parts = append(parts, providers.ContentPart{Type: providers.PartText, Text: prompt})
	}
	return providers.Message{Role: providers.RoleUser, Content: parts}, nil
}

func looksLikeText(b []byte) bool {
	// Reject if NUL byte appears in the first 8KB.
	end := len(b)
	if end > 8192 {
		end = 8192
	}
	for _, c := range b[:end] {
		if c == 0 {
			return false
		}
	}
	return true
}

func mimeFromExt(path string) string {
	ext := strings.ToLower(filepath.Ext(path))
	switch ext {
	case ".png":
		return "image/png"
	case ".jpg", ".jpeg":
		return "image/jpeg"
	case ".webp":
		return "image/webp"
	case ".gif":
		return "image/gif"
	}
	return "application/octet-stream"
```

```go
type Meta struct {
	ID       string    `json:"id"`
	Created  time.Time `json:"created"`
	Provider string    `json:"provider"`
	Model    string    `json:"model"`
	Profile  string    `json:"profile,omitempty"`
	Title    string    `json:"title,omitempty"`

type Conversation struct {
	Meta     Meta
	Messages []providers.Message
}

// Init creates a new conversation file with a meta line. Fails if the file
// already exists and has content.
func Init(path string, meta Meta) error {
	if meta.Created.IsZero() {
		meta.Created = time.Now().UTC()
	}
	f, err := os.OpenFile(path, os.O_WRONLY|os.O_CREATE|os.O_EXCL, 0o600)
	if err != nil {
		return err
	}
	defer f.Close()
	return writeLine(f, map[string]any{
		"type": "meta", "id": meta.ID, "created": meta.Created,
		"provider": meta.Provider, "model": meta.Model,
		"profile": meta.Profile, "title": meta.Title,
	})
}

// Append adds a message.
func Append(path string, m providers.Message) error {
	f, err := os.OpenFile(path, os.O_WRONLY|os.O_APPEND|os.O_CREATE, 0o600)
	if err != nil {
		return err
	}
	defer f.Close()
	entry := struct {
		Type string `json:"type"`
		providers.Message
	}{"message", m}
	return writeLine(f, entry)
}

// Read loads the full conversation.
func Read(path string) (Conversation, error) {
	f, err := os.Open(path)
	if err != nil {
		return Conversation{}, err
	}
	defer f.Close()

	var c Conversation
	sc := bufio.NewScanner(f)
	sc.Buffer(make([]byte, 1<<16), 1<<24)
	first := true
	for sc.Scan() {
		line := sc.Bytes()
		if first {
			first = false
			var m struct {
				Type string `json:"type"`
				Meta
			}
			if err := json.Unmarshal(line, &m); err != nil {
				return Conversation{}, fmt.Errorf("meta line: %w", err)
			}
			if m.Type != "meta" {
				return Conversation{}, fmt.Errorf("first line is not meta")
			}
			c.Meta = m.Meta
			continue
		}
		var env struct {
			Type string `json:"type"`
			providers.Message
		}
		if err := json.Unmarshal(line, &env); err != nil {
			continue // skip corrupt lines
		}
		if env.Type == "message" {
			c.Messages = append(c.Messages, env.Message)
		}
	}
	return c, sc.Err()
}

// Exists reports whether the path exists.
func Exists(path string) bool {
	_, err := os.Stat(path)
	return err == nil || !errors.Is(err, fs.ErrNotExist)
}

func writeLine(f *os.File, v any) error {
	b, err := json.Marshal(v)
	if err != nil {
		return err
	}
	if _, err := f.Write(append(b, '\n')); err != nil {
		return err
	}
	return nil
}
```

```go
func emitFromResponse(resp providers.Response, f output.Format, w interface{ Write([]byte) (int, error) }) error {
    ch := make(chan providers.Chunk, 8)
    go func() {
        defer close(ch)
        for _, m := range resp.Messages {
            for _, part := range m.Content {
                if part.Type == providers.PartText && part.Text != "" {
                    ch <- providers.Chunk{Type: providers.ChunkTextDelta, Text: part.Text}
                }
            }
            for _, tc := range m.ToolCalls {
                tc := tc
                ch <- providers.Chunk{Type: providers.ChunkToolCallStart, ToolCall: &tc}
            }
        }
        ch <- providers.Chunk{Type: providers.ChunkUsage, Usage: &resp.Usage}
        ch <- providers.Chunk{Type: providers.ChunkMessageStop}
    }()
    wr := output.NewWriter(f, w)
    if err := wr.Consume(ch); err != nil {
        return err
    }
    return wr.Finish()
}

func runStreamAndPersist(ctx context.Context, p providers.Provider, req providers.Request, f output.Format,
    w interface{ Write([]byte) (int, error) }, convPath string, userMsg providers.Message) error {
    req.Stream = true
    ch, err := p.Stream(ctx, req)
    if err != nil {
        return err
    }
    wr := output.NewWriter(f, w)
    var collected strings.Builder
    var tools []providers.ToolCall
    for c := range ch {
        switch c.Type {
        case providers.ChunkTextDelta:
            collected.WriteString(c.Text)
        case providers.ChunkToolCallStart:
            if c.ToolCall != nil {
                tools = append(tools, *c.ToolCall)
            }
        case providers.ChunkError:
            return c.Err
        }
        // Re-emit to the formatter by creating a single-chunk channel.
        sub := make(chan providers.Chunk, 1)
        sub <- c
        close(sub)
        if err := wr.Consume(sub); err != nil {
            return err
        }
    }
    if convPath != "" {
        _ = chat.Append(convPath, userMsg)
        _ = chat.Append(convPath, providers.Message{
            Role: providers.RoleAssistant,
            Content: []providers.ContentPart{{Type: providers.PartText, Text: collected.String()}},
            ToolCalls: tools,
        })
    }
    return wr.Finish()
}

func randID() string {
    return strings.ReplaceAll(time.Now().UTC().Format("20060102T150405.000000"), ".", "") +
        "-" + strconv.Itoa(int(time.Now().UnixNano()%1e6))
}
```

```go
package chat

import "strings"

type SlashCommand struct {
	Name    string // leading slash included, e.g. "/model"
	Summary string
}

// DefaultSlashCommands returns the builtin set.
func DefaultSlashCommands() []SlashCommand {
	return []SlashCommand{
		{"/model", "change model for this session"},
		{"/provider", "change provider for this session"},
		{"/system", "set system prompt for this session"},
		{"/save", "save the conversation to disk"},
		{"/clear", "start a new conversation"},
		{"/exit", "exit the REPL"},
		{"/help", "show available commands"},
	}
}

// CompleteSlash returns commands matching the given prefix (case-insensitive).
// Returns nil if input doesn't start with '/'.
func CompleteSlash(input string, all []SlashCommand) []SlashCommand {
	if !strings.HasPrefix(input, "/") {
		return nil
	}
	var out []SlashCommand
	for _, c := range all {
		if strings.HasPrefix(strings.ToLower(c.Name), strings.ToLower(input)) {
			out = append(out, c)
		}
	}
	return out
}
```

```go
// input on Enter (Shift+Enter inserts a newline on supporting terminals).

func ReadInput(prompt string, commands []SlashCommand) (string, error) {

	ti.SetHeight(3)
	ti.Focus()

	m := &replModel{ta: ti, prompt: prompt, cmds: commands}
	p := tea.NewProgram(m, tea.WithAltScreen())
	if _, err := p.Run(); err != nil {
		return "", err
	}
	if m.quit {
		return "", fmt.Errorf("exit")
	}
	return strings.TrimRight(m.submitted, "\n"), nil
}

type replModel struct {
	ta        textarea.Model
	prompt    string
	cmds      []SlashCommand
	submitted string
	quit      bool
}

func (m *replModel) Init() tea.Cmd {
	return textarea.Blink
}

func (m *replModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.Type {
		case tea.KeyCtrlD, tea.KeyCtrlC:
			m.quit = true
			return m, tea.Quit
		case tea.KeyEnter:
			// Enter (without shift) submits.
			if !msg.Alt && !keyHasShift(msg) {
				m.submitted = m.ta.Value()
				if strings.TrimSpace(m.submitted) == "/exit" {
					m.quit = true
				}
				return m, tea.Quit
			}
		case tea.KeyTab:
			if sugg := CompleteSlash(m.ta.Value(), m.cmds); len(sugg) == 1 {
				m.ta.SetValue(sugg[0].Name + " ")
				return m, nil
			}
		}
	}
	var cmd tea.Cmd
	m.ta, cmd = m.ta.Update(msg)
	return m, cmd
}

func (m *replModel) View() string {
	header := lipgloss.NewStyle().Foreground(lipgloss.Color("6")).Render(m.prompt)
	return header + "\n" + m.ta.View()
}

// keyHasShift checks for Shift+Enter where the terminal reports modifiers.
// bubbletea surfaces these via msg.Alt for Alt; shift-on-enter currently
// arrives as a distinct KeyType depending on terminal. This helper is a
// single tweakable point when we add better modifier detection.
func keyHasShift(msg tea.KeyMsg) bool {
	// bubbletea v0.25+ exposes key.Modifier; for now rely on the fact that
	// shift+enter on kitty/wezterm sends \x1b[13;2u which bubbletea maps to
	// tea.KeyShiftEnter (if defined) or leaves as unknown — treat unknowns
	// as plain Enter.
	return false
}
```

```go
// List returns metas of every conversation in dir, newest first. Only the
// first line of each file is parsed.
func List(dir string) ([]Meta, error) {
	entries, err := os.ReadDir(dir)
	if err != nil {
		return nil, err
	}
	out := []Meta{}
	for _, e := range entries {
		if e.IsDir() || !strings.HasSuffix(e.Name(), ".jsonl") {
			continue
		}
		p := filepath.Join(dir, e.Name())
		meta, err := readMeta(p)
		if err != nil {
			continue
		}
		out = append(out, meta)
	}
	sort.Slice(out, func(i, j int) bool { return out[i].Created.After(out[j].Created) })
	return out, nil
}

func readMeta(path string) (Meta, error) {
	f, err := os.Open(path)
	if err != nil {
		return Meta{}, err
	}
	defer f.Close()
	sc := bufio.NewScanner(f)
	if !sc.Scan() {
		return Meta{}, fmt.Errorf("empty file")
	}
	var m struct {
		Type string `json:"type"`
		Meta
	}
	if err := json.Unmarshal(sc.Bytes(), &m); err != nil {
		return Meta{}, err
	}
	return m.Meta, nil
}

func shortRand() string {
	const chars = "abcdef0123456789"
	var raw [3]byte
	if _, err := rand.Read(raw[:]); err != nil {
		// Fall back to nanosecond mixing; still unlikely to collide in
		// normal interactive use.
		n := time.Now().UnixNano()
		raw[0] = byte(n)
		raw[1] = byte(n >> 8)
		raw[2] = byte(n >> 16)
	}
	var out [6]byte
	for i, b := range raw {
		out[i*2] = chars[b>>4]
		out[i*2+1] = chars[b&0x0f]
	}
	return string(out[:])
}
```

```go
func runReplLoop(ctx context.Context, p providers.Provider, r config.Resolved, convPath string, w interface{ Write([]byte) (int, error) }) error {
	conv, err := chat.Read(convPath)
	if err != nil {
		return err
	}
	history := conv.Messages
	system := r.System

	cmds := chat.DefaultSlashCommands()

	for {
		input, err := chat.ReadInput(fmt.Sprintf("ai (%s · %s)", r.ProviderName, r.Model), cmds)
		if err != nil {
			return nil // exit
		}
		if strings.HasPrefix(input, "/") {
			switch fields := strings.Fields(input); fields[0] {
			case "/exit":
				return nil
			case "/clear":
				history = nil
				continue
			case "/model":
				if len(fields) >= 2 {
					r.Model = fields[1]
				}
				continue
			case "/provider":
				if len(fields) >= 2 {
					r.ProviderName = fields[1]
					// Rebuilding the provider is left to the engineer; for v1
					// we just update the label and new messages still route
					// through the original provider. The spec allows this as
					// a simple v1 scope.
				}
				continue
			case "/system":
				system = strings.TrimSpace(strings.TrimPrefix(input, "/system"))
				continue
			case "/help":
				for _, c := range cmds {
					fmt.Fprintf(w, "%s\t%s\n", c.Name, c.Summary)
				}
				continue
			}
		}

		userMsg := providers.Message{
			Role: providers.RoleUser,
			Content: []providers.ContentPart{{Type: providers.PartText, Text: input}},
		}
		history = append(history, userMsg)
		_ = chat.Append(convPath, userMsg)

		req := providers.Request{Model: r.Model, System: system, Messages: history, Stream: true}
		ch, err := p.Stream(ctx, req)
		if err != nil {
			fmt.Fprintln(w, "error:", err)
			continue
		}
		var collected strings.Builder
		for c := range ch {
			switch c.Type {
			case providers.ChunkTextDelta:
				fmt.Fprint(w, c.Text)
				collected.WriteString(c.Text)
			case providers.ChunkError:
				fmt.Fprintln(w, "\nerror:", c.Err)
			}
		}
		fmt.Fprintln(w)

		assistant := providers.Message{
			Role: providers.RoleAssistant,
			Content: []providers.ContentPart{{Type: providers.PartText, Text: collected.String()}},
		}
		history = append(history, assistant)
		_ = chat.Append(convPath, assistant)
	}
}

func findByPrefix(dir, prefix string) (string, error) {
	entries, err := os.ReadDir(dir)
	if err != nil {
		return "", err
	}
	var matches []string
	for _, e := range entries {
		if !strings.HasSuffix(e.Name(), ".jsonl") {
			continue
		}
		id := strings.TrimSuffix(e.Name(), ".jsonl")
		if strings.HasPrefix(id, prefix) {
			matches = append(matches, filepath.Join(dir, e.Name()))
		}
	}
	switch len(matches) {
	case 0:
		return "", fmt.Errorf("no conversation matches %q", prefix)
	case 1:
		return matches[0], nil
	default:
		return "", fmt.Errorf("ambiguous prefix %q (%d matches)", prefix, len(matches))
	}
}
```

```go
func buildBinary(t *testing.T) string {
	t.Helper()
	root, err := os.Getwd()
	require.NoError(t, err)
	// Walk up to repo root (where go.mod lives).
	for {
		if _, err := os.Stat(filepath.Join(root, "go.mod")); err == nil {
			break
		}
		parent := filepath.Dir(root)
		if parent == root {
			t.Fatal("go.mod not found")

		root = parent
	}
	bin := filepath.Join(root, "bin", "ai")
	if runtime.GOOS == "windows" {
		bin += ".exe"
	}
	cmd := exec.Command("go", "build", "-o", bin, "./cmd/ai")
	cmd.Dir = root
	out, err := cmd.CombinedOutput()
	require.NoError(t, err, "build failed: %s", out)
	return bin
}

func run(t *testing.T, bin string, args ...string) (stdout, stderr string, exitCode int) {
	t.Helper()
	var so, se bytes.Buffer

	c.Stderr = &se

		exitCode = ee.ExitCode()
	} else if err != nil {
		t.Fatalf("run: %v", err)
	}
	return so.String(), se.String(), exitCode
```

```go
package cli

import "errors"

// Exit codes per spec §6.2.
const (
	ExitOK          = 0
	ExitAPI         = 1
	ExitUsage       = 2
	ExitAuth        = 3
	ExitInput       = 4
	ExitInterrupted = 130
)

type exitError struct {
	code int
	err  error
}

func (e *exitError) Error() string { return e.err.Error() }
func (e *exitError) Unwrap() error { return e.err }
func (e *exitError) ExitCode() int { return e.code }

// NewExitError wraps an error with a specific exit code.
func NewExitError(code int, err error) error {
	if err == nil {
		return nil
	}
	return &exitError{code: code, err: err}
}

// ExitCodeOf returns the exit code for any error. Unwrapped errors get
// ExitAPI (1), which is the default non-success.
func ExitCodeOf(err error) int {
	if err == nil {
		return ExitOK
	}
	var e *exitError
	if errors.As(err, &e) {
		return e.code
	}
	return ExitAPI
}
```

Append under `tasks:`:

```yaml
  release:ci:
    desc: Cross-compile all release artifacts to ./dist
    cmds:
      - mkdir -p ./dist
      - |
        for os in linux darwin windows; do
          case "$os" in
            linux) arches="amd64 arm64"; ext=""; ;;
            darwin) arches="amd64 arm64"; ext=""; ;;
            windows) arches="amd64"; ext=".exe"; ;;
          esac
          for arch in $arches; do
            base="ai_{{.VERSION}}_${os}_${arch}"
            GOOS=$os GOARCH=$arch CGO_ENABLED=0 go build -trimpath \
              -ldflags="{{.LDFLAGS}}" -o dist/$base/ai$ext ./cmd/ai
            tar -C dist -czf dist/$base.tar.gz $base
          done
        done
      - |
        # Linux static variants (already static via CGO_ENABLED=0 above, but
        # produce a second tarball with the "_static" suffix for clarity).
        for arch in amd64 arm64; do
          base="ai_{{.VERSION}}_linux_${arch}"
          cp dist/$base/ai dist/${base}_static/ai
          mkdir -p dist/${base}_static
          cp dist/$base/ai dist/${base}_static/ai
          tar -C dist -czf dist/${base}_static.tar.gz ${base}_static
        done
      - cd dist && sha256sum *.tar.gz > checksums.txt
```
